function test(endpoint, data) {
	let xhr = new XMLHttpRequest();
	
	if (data) {
		xhr.open('POST', endpoint);
		xhr.send(JSON.stringify(data));
	} else {
		xhr.open('GET', endpoint);
		xhr.send();
	}
}
